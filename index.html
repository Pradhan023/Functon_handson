Q7. Write a function that accepts parameter n and returns factorial of n

        
Ans. const fact = (n) => 
            {
                let num =1
                for(let i = 1 ; i <= n ; i++)
                {
                    num = num * i;
                }
                return num
            }
            console.log(fact(5));
            OUTPUT: 120
